Sub MacroCalPorc1()
'
' Realiza el primer calculo de los porcentajes
'

'
    ActiveCell.FormulaR1C1 = "=RC[-9]+R[2]C[-2]"
    Range("J7").Select
    ActiveCell.FormulaR1C1 = "=RC[-9]+RC[-2]"
    Range("J8").Select
    ActiveCell.FormulaR1C1 = "=RC[-9]+R[-2]C[-2]"
    Range("K6").Select
    ActiveCell.FormulaR1C1 = "=RC[-1]/2"
    Range("K6").Select
    Selection.AutoFill Destination:=Range("K6:K8"), Type:=xlFillDefault
    Range("K6:K8").Select
    Range("I12").Select
    ActiveCell.FormulaR1C1 = "L"
    Range("I13").Select
    ActiveCell.FormulaR1C1 = "V"
    Range("J12").Select
    ActiveCell.FormulaR1C1 = "=RC[-9]-R[1]C[-9]"
    Range("J13").Select
    ActiveCell.FormulaR1C1 = "=R[-1]C[-5]-RC[-5]"
    Range("J14").Select
    ActiveCell.FormulaR1C1 = "=R[-2]C-R[-1]C"
    Range("J14").Select
End Sub


Sub MixPorcGSupremacy()
'
' Realiza el calculo final de las probabilidades, hace el mix entre los porcentajes
' de la subrutina anterior con los porcentajes dados por el G Supremacy
'

'
    ActiveCell.FormulaR1C1 = "=RC[-1]+RC[-3]"
    Range("N6").Select
    Selection.AutoFill Destination:=Range("N6:N8"), Type:=xlFillDefault
    Range("N6:N8").Select
    Range("O6").Select
    ActiveCell.FormulaR1C1 = "=RC[-1]/2"
    Range("O6").Select
    Selection.AutoFill Destination:=Range("O6:O8"), Type:=xlFillDefault
    Range("O6:O8").Select
    Range("P6").Select
    ActiveCell.FormulaR1C1 = "L"
    Range("P7").Select
    ActiveCell.FormulaR1C1 = "X"
    Range("P8").Select
    ActiveCell.FormulaR1C1 = "V"
    Range("O6:P8").Select
    With Selection.Interior
        .Pattern = xlSolid
        .PatternColorIndex = xlAutomatic
        .Color = 65535

    Selection.Borders(xlDiagonalDown).LineStyle = xlNone
    Selection.Borders(xlDiagonalUp).LineStyle = xlNone
    With Selection.Borders(xlEdgeLeft)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0

    End With
    With Selection.Borders(xlEdgeTop)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlEdgeBottom)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlEdgeRight)
        .LineStyle = xlContinuous
        .ColorIndex = 0
        .TintAndShade = 0
        .Weight = xlThin
    End With
    With Selection.Borders(xlInsideVertical)
        .LineStyle = xlContinuous
        .ColorIndex = 0

    End With
    With Selection.Borders(xlInsideHorizontal)
